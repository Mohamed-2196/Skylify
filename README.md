# React + Vite + Python 3.10.11
# steps to run project:
    1. npm install
    2. npm run dev
    3. python -m venv env_name
    4. source env_name/bin/activate
    5. add env name to .gitignore
    6. pip install -r requirements.txt
    7. cd python_backend/
    8. uvicorn main:app --reload
